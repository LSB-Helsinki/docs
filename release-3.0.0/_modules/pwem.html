    def defineBinariesMaxit(cls, default, env):
        MAXIT_URL = 'https://sw-tools.rcsb.org/apps/MAXIT/maxit-v10.100-prod-src.tar.gz'
        MAXIT_TAR = 'maxit-v10.100-prod-src.tar.gz'

        env.addPackage(MAXIT, version='10.1',

                       url=MAXIT_URL,
